三隻小豬

41023235所作:

網站:http://127.0.0.1:5000/game
成果影片:https://www.youtube.com/watch?v=OzSOG_ulA78
可在瀏覽器執行、畫面不空白、只有三隻豬、永不重疊的gmme.py:
# game.py
from browser import document, html, timer, ajax, window
from random import random

canvas = document["gameCanvas"]
ctx = canvas.getContext("2d")
WIDTH, HEIGHT = 800, 400

# 圖片
bird_img = html.IMG(src="./../cmsimde/static/images/bird.png")
pig_img = html.IMG(src="./../cmsimde/static/images/pig.png")

# 遊戲狀態
SLING_X, SLING_Y = 120, 300
MAX_SHOTS = 10
shots_fired = 0
total_score = 0
current_shot_score = 0
mouse_down = False
mouse_pos = (SLING_X, SLING_Y)
projectile = None
sent = False

# ------------------------------------------
# 類別
# ------------------------------------------
class Pig:
def __init__(self,x,y):
self.x,self.y = x,y
self.w,self.h = 40,40
self.alive = True

# 房舍相對於小豬的相對位置 (整體用來碰撞檢查)
self.house_blocks = [
(0, 40, 120, 15), # 地基
(0, -10, 15, 50), # 左牆
(105, -10, 15, 50), # 右牆
(0, -25, 120, 15) # 屋頂
]
# 計算整體矩形範圍
self.compute_bounds()

def compute_bounds(self):
# 找到房子和豬的最大矩形範圍
xs = [rx-40 for rx,_,rw,_ in self.house_blocks] + [0]
ys = [ry for _,ry,_,rh in self.house_blocks] + [0]
ws = [rw for _,_,rw,_ in self.house_blocks] + [self.w]
hs = [rh for _,_,_,rh in self.house_blocks] + [self.h]
min_x = min(xs)
min_y = min(ys)
max_x = max([x+w for x,w in zip(xs,ws)])
max_y = max([y+h for y,h in zip(ys,hs)])
self.bound_x = self.x + min_x
self.bound_y = self.y + min_y
self.bound_w = max_x - min_x
self.bound_h = max_y - min_y

def draw(self):
if self.alive:
ctx.fillStyle="saddlebrown"
for rx, ry, rw, rh in self.house_blocks:
ctx.fillRect(self.x + rx - 40, self.y + ry, rw, rh)
ctx.drawImage(pig_img,self.x,self.y,self.w,self.h)
self.compute_bounds()

def hit(self,px,py):
return self.alive and self.x<=px<=self.x+self.w and self.y<=py<=self.y+self.h

def relocate(self, all_pigs):
MIN_X = 450
MAX_X = WIDTH - self.w - 120
MIN_Y = 200
MAX_Y = HEIGHT - self.h - 15

attempts = 0
while True:
new_x = MIN_X + random() * (MAX_X - MIN_X)
new_y = MIN_Y + random() * (MAX_Y - MIN_Y)
# 更新暫時位置，用於矩形碰撞檢查
temp_bound_x = new_x - 40
temp_bound_y = new_y - 25
temp_bound_w = 120
temp_bound_h = 75

# 檢查與其他豬房屋矩形是否重疊
overlap = False
for p in all_pigs:
if p == self:
continue
if (temp_bound_x < p.bound_x + p.bound_w and
temp_bound_x + temp_bound_w > p.bound_x and
temp_bound_y < p.bound_y + p.bound_h and
temp_bound_y + temp_bound_h > p.bound_y):
overlap = True
break
if not overlap:
self.x = new_x
self.y = new_y
self.compute_bounds()
break
attempts += 1
if attempts > 1000:
break

class Bird:
def __init__(self,x,y,vx,vy):
self.x,self.y,self.vx,self.vy = x,y,vx,vy
self.w,self.h = 35,35
self.active = True

def update(self):
global current_shot_score, total_score
if not self.active: return
self.vy += 0.35
self.x += self.vx
self.y += self.vy

if self.y > HEIGHT - self.h or self.x > WIDTH or self.x < 0:
self.active = False

for p in pigs:
if p.hit(self.x+self.w/2,self.y+self.h/2):
if p.alive:
p.relocate(pigs)
current_shot_score += 50
total_score += 50
document["score_display"].text = str(total_score)

def draw(self):
ctx.drawImage(bird_img,self.x,self.y,self.w,self.h)

# ------------------------------------------
# 遊戲控制函數
# ------------------------------------------
pigs = []

def init_level():
global pigs
pigs = []
positions = []
MIN_X = 450
MAX_X = WIDTH - 40 - 120
MIN_Y = 200
MAX_Y = HEIGHT - 40 - 15
while len(pigs) < 3:
x = MIN_X + random() * (MAX_X - MIN_X)
y = MIN_Y + random() * (MAX_Y - MIN_Y)
# 房屋整體矩形
bound_x = x - 40
bound_y = y - 25
bound_w = 120
bound_h = 75
# 檢查與已生成的豬房屋不重疊
overlap = False
for px, py, pw, ph in positions:
if (bound_x < px + pw and bound_x + bound_w > px and
bound_y < py + ph and bound_y + bound_h > py):
overlap = True
break
if not overlap:
pig = Pig(x, y)
pigs.append(pig)
positions.append((bound_x, bound_y, bound_w, bound_h))

def reset_sling_state():
global shots_fired, current_shot_score, projectile, sent
shots_fired = 0
current_shot_score = 0
projectile = None
sent = False
init_level()
update_shots_remaining()

def start_new_game():
global total_score
total_score = 0
document["score_display"].text = str(total_score)
reset_sling_state()

def update_shots_remaining():
document["shots_remaining"].text = str(MAX_SHOTS - shots_fired)

def get_mouse_pos(evt):
return evt.x - canvas.offsetLeft, evt.y - canvas.offsetTop

def mousedown(evt):
global mouse_down, mouse_pos
if projectile is None and shots_fired < MAX_SHOTS:
mouse_down = True
mouse_pos = get_mouse_pos(evt)

def mousemove(evt):
global mouse_pos
if mouse_down:
mouse_pos = get_mouse_pos(evt)

def mouseup(evt):
global mouse_down, projectile, shots_fired, current_shot_score
if mouse_down and projectile is None:
current_shot_score = 0
mouse_down = False
end_pos = get_mouse_pos(evt)
dx = SLING_X - end_pos[0]
dy = SLING_Y - end_pos[1]
projectile = Bird(SLING_X, SLING_Y, dx*0.25, dy*0.25)
shots_fired += 1
update_shots_remaining()

canvas.bind("mousedown", mousedown)
canvas.bind("mousemove", mousemove)
canvas.bind("mouseup", mouseup)

def draw_sling():
ctx.strokeStyle="black"
ctx.lineWidth=4
if mouse_down:
mx,my = mouse_pos
ctx.beginPath()
ctx.moveTo(SLING_X-5, SLING_Y)
ctx.lineTo(mx,my)
ctx.stroke()
ctx.beginPath()
ctx.moveTo(SLING_X+5, SLING_Y)
ctx.lineTo(mx,my)
ctx.stroke()
ctx.drawImage(bird_img,mx-17,my-17,35,35)
else:
if projectile is None and shots_fired < MAX_SHOTS:
ctx.drawImage(bird_img,SLING_X-17,SLING_Y-17,35,35)

def send_score():
global sent, total_score
if sent: return
sent = True
def on_complete(req):
if req.status == 200:
print("Score saved successfully.")
else:
try:
resp = window.JSON.parse(req.text)
msg = resp.message
except:
msg = "Server error or bad response format."
print(f"Score submission failed: HTTP {req.status}, Message: {msg}")
req = ajax.ajax()
req.bind('complete', on_complete)
req.open("POST","/submit_score", True)
req.set_header("Content-Type","application/json")
data = {"score":total_score}
json_payload = window.JSON.stringify(data)
req.send(json_payload)

# ------------------------------------------
# 遊戲主迴圈
# ------------------------------------------
def loop():
global projectile
ctx.clearRect(0,0,WIDTH,HEIGHT)

for p in pigs: p.draw()

game_over = shots_fired >= MAX_SHOTS

if projectile:
projectile.update()
projectile.draw()

if not projectile.active:
projectile = None
if game_over:
send_score()
timer.set_timeout(start_new_game, 2000)

elif game_over:
send_score()
timer.set_timeout(start_new_game, 2000)

draw_sling()

timer.set_interval(loop,30)
start_new_game()

課程學習心得（41023235 黃俊錡）

透過本課程的學習與實作，我對電腦程式設計在互動模擬與系統行為控制上的應用有了更深入的理解。課程中不只是學習程式語法，而是透過實際撰寫與修改程式，將抽象的邏輯轉換成可在畫面中即時呈現的互動結果。

在實作「三隻小豬」的遊戲過程中，我實際遇到了多項困難。首先是在關卡初始化時，小豬常常生成在相同或非常接近的位置，導致畫面重疊，甚至在被擊中時出現「同歸於盡」的情況。為了解決這個問題，我嘗試加入隨機生成座標的機制，並透過設定最小距離，逐一檢查新生成的位置是否與已存在的小豬過於接近，確保三隻小豬能夠分散且不重疊地出現在畫面中。

另一個困難是在小豬被擊中後的行為設計。一開始小豬被擊中後，位置沒有重新調整，導致仍然集中在同一區域，影響遊戲的可玩性。因此，我進一步調整程式邏輯，讓小豬在被擊中後能重新隨機配置位置，並再次檢查彼此之間的距離，使遊戲流程更加合理且穩定。

在反覆修改與測試的過程中，我也曾遇到程式錯誤導致網頁畫面全白、遊戲無法執行的情況，這讓我體會到程式結構與函式修改必須謹慎，任何一個小錯誤都可能影響整個系統運作。透過不斷檢查錯誤訊息與調整程式內容，我逐漸培養了除錯與問題分析的能力。

透過這些實作經驗，我了解到程式設計不只是讓功能完成，更需要考慮物件之間的互動關係、系統穩定性以及使用者體驗。這樣的概念在遊戲設計、模擬系統與工程應用中都相當重要。

此外，透過 pygame 這類圖形與事件導向的程式設計方式，我對互動式系統的運作有更清楚的概念，例如使用者輸入如何影響系統狀態、物件如何依據規則即時回應。這樣的能力不僅限於遊戲，也能延伸應用到人機介面設計、機器人模擬、智慧裝置控制與教學模擬系統等領域。

就未來應用而言，學生可以將這些技能運用在以下方向：

  1. 工程與機械相關領域：用於機構運動模擬、碰撞檢測、系統行為預測。

  2. 遊戲與互動媒體設計：設計具有規則判斷與即時反應的互動內容。

  3. 機器人與自動化模擬：在實體製作前，先以程式模擬行為與環境互動。

  4. 跨領域專題與專案實作：結合設計、工程與程式能力，提升專題完整度。

整體而言，這門課讓我體會到程式設計在實務上的價值，也培養了我將問題拆解、設計邏輯並逐步除錯的能力。這些能力不僅有助於後續課程學習，也能成為未來進入職場或進行專題研究時的重要基礎。
